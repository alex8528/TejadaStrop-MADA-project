04 Statistical Analysis

Author

Alexandra Tejada-Strop

This page performs the main statistical analyses for the project.

It fits a bivariate logistic regression model examining the association between prenatal mental health screening and postpartum depressive symptoms, followed by a multivariable logistic regression adjusting for selected covariates. A tidymodels workflow is also fit and saved for downstream use.

library(here)
library(tidymodels)
library(broom)

# Define variables
outcome <- "pp_depress_bin"
main_exposure <- "screened"
covariates <- c("bpg_deprs8", "mat_ed")

all_vars <- c(outcome, main_exposure, covariates)
stopifnot(all(all_vars %in% names(dat)))

# Create complete-case analysis dataset
analysis_dat <- dat |>
  select(all_of(all_vars)) |>
  drop_na() |>
  mutate(
    screened = as.factor(screened),
    bpg_deprs8 = as.factor(bpg_deprs8),
    mat_ed = as.factor(mat_ed)
  )

Tidymodels expects a factor outcome for classification

analysis_dat2 <- analysis_dat |>
  mutate(pp_depress_bin = factor(pp_depress_bin, levels = c(0, 1)))

rec <- recipe(pp_depress_bin ~ screened + bpg_deprs8 + mat_ed, data = analysis_dat2) |>
  step_dummy(all_nominal_predictors()) |>
  step_zv(all_predictors())

wf <- workflow() |>
  add_recipe(rec) |>
  add_model(logistic_reg() |> set_engine("glm"))

fit_wf <- fit(wf, data = analysis_dat2)

Save tidymodels workflow

saveRDS(fit_wf, here("results", "output", "tidymodels_logistic_main.rds"))
