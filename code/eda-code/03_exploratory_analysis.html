Plot: Postpartum depressive symptoms by screening

data_summary <- analytic_dat |>
  count(screened, pp_depress_bin) |>
  group_by(screened) |>
  mutate(proportion = n / sum(n)) |>
  mutate(
    screening_status = case_when(
      screened == 0 ~ "Not screened",
      screened == 1 ~ "Screened",
      TRUE ~ as.character(screened)
    ),
    depression_status = case_when(
      pp_depress_bin == 0 ~ "No depressive symptoms",
      pp_depress_bin == 1 ~ "Postpartum depressive symptoms",
      TRUE ~ as.character(pp_depress_bin)
    )
  )

plot_screen_dep <- ggplot(
  data_summary,
  aes(
    x = screening_status,
    y = proportion,
    fill = depression_status
  )
) +
  geom_col(width = 0.65, color = "white") +
  geom_text(
    aes(label = scales::percent(proportion, accuracy = 0.1)),
    position = position_stack(vjust = 0.5),
    size = 4
  ) +
  scale_y_continuous(
    labels = scales::percent,
    expand = expansion(mult = c(0, .05))
  ) +
  scale_fill_manual(
    values = c(
      "No depressive symptoms" = "#4E79A7",
      "Postpartum depressive symptoms" = "#E15759"
    )
  ) +
  labs(
    title = "Postpartum Depressive Symptoms by Prenatal Mental Health Screening",
    subtitle = "PRAMS Phase 8",
    x = "Prenatal mental health screening",
    y = "Percent of respondents",
    fill = "Postpartum depressive symptoms"
  ) +
  theme_minimal(base_size = 13) +
  theme(
    legend.position = "top",
    plot.title = element_text(face = "bold", size = 14),
    axis.title = element_text(face = "bold")
  )

Cross tabulation

cross_tab <- analytic_dat |>
  count(screened, pp_depress_bin) |>
  group_by(screened) |>
  mutate(
    percent = round(n / sum(n) * 100, 1),
    screening_status = case_when(
      screened == 0 ~ "Not screened",
      screened == 1 ~ "Screened",
      TRUE ~ as.character(screened)
    ),
    depression_status = case_when(
      pp_depress_bin == 0 ~ "No depressive symptoms",
      pp_depress_bin == 1 ~ "Postpartum depressive symptoms",
      TRUE ~ as.character(pp_depress_bin)
    )
  ) |>
  select(screening_status, depression_status, n, percent)

Interpretation

The exploratory analysis suggests that postpartum depressive symptoms were slightly less common among respondents who reported being screened during prenatal care. About 13.9% of screened respondents reported postpartum depressive symptoms compared with 15.2% among those who were not screened.

Save screening summary table

screening_table <- analytic_dat |>
  count(screened, pp_depress_bin) |>
  group_by(screened) |>
  mutate(percent = round(n / sum(n) * 100, 1)) |>
  mutate(
    `Prenatal screening` = case_when(
      screened == 0 ~ "Not screened",
      screened == 1 ~ "Screened",
      TRUE ~ as.character(screened)
    ),
    `Postpartum depressive symptoms` = case_when(
      pp_depress_bin == 0 ~ "No",
      pp_depress_bin == 1 ~ "Yes",
      TRUE ~ as.character(pp_depress_bin)
    )
  ) |>
  select(
    `Prenatal screening`,
    `Postpartum depressive symptoms`,
    n,
    percent
  )
